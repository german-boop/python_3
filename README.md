# Simple Graph Example

import matplotlib.pyplot as plt  # Import matplotlib for plotting

# Sample data
x = [1, 2, 3, 4, 5]
y = [2, 4, 6, 8, 10]

# Plot the data
plt.plot(x, y, marker='o')  # Line plot with circles on points

# Add title and labels
plt.title("Simple Graph")
plt.xlabel("X-axis")
plt.ylabel("Y-axis")

# Show the graph
plt.show()
